const HomePage: React.FC = () => {
  const [ads, setAds] = useState<Ad[]>([]);
  const [loadingAds, setLoadingAds] = useState(true);
  const [filteredAds, setFilteredAds] = useState<Ad[]>([]);
  const navigate = useNavigate();
  const location = useLocation();
  
 
  useEffect(() => {
    const loadAds = async () => {
      try {
        setLoadingAds(true);
        const allAds = await adService.getAllAds();
        setAds(allAds);
      } catch (error) {
        console.error('Error loading ads:', error);
      } finally {
        setLoadingAds(false);
      }
    };
 
    loadAds();
  }, []);
 
  // Filter ads - immer Kleinanzeigen für die Hauptseite
  useEffect(() => {
    // Zeige nur Kleinanzeigen (alle außer Autos) auf der Hauptseite
    setFilteredAds(ads.filter(ad => 
      ad.category?.toLowerCase() !== 'autos'
    ));
  }, [ads]);
 
  return (
    <Box sx={{ 
      width: '100%', 
      py: { xs: 0.5, sm: 0.5, md: 0.5 }, // Reduziert von 1/1/1 auf 0.5/0.5/0.5
      height: '100%',
      bgcolor: '#ffffff'
    }}>
      {/* Keine Suchmaske mehr - direkt zu Kleinanzeigen-Inhalten */}
      <Box sx={{ px: { xs: 1, sm: 2, md: 3 }, bgcolor: '#ffffff', pt: { xs: 0.5, sm: 1, md: 1 } }}>
        {/* Kategorie-Buttons direkt über den Kategorie-Karten */}
        <Box sx={{ 
          display: 'flex', 
          justifyContent: 'center', 
          gap: 2, 
          mb: { xs: 0.5, md: 1 }, // Weiter reduziert von 1/1.5 auf 0.5/1
          flexWrap: 'wrap'
        }}>
          <Button
            variant="text"
            onClick={() => navigate('/category/kleinanzeigen')}
            startIcon={
              <Box
                component="img"
                src="/images/kleinanzeigen_icon.webp"
                alt="Kleinanzeigen"
                sx={{
                  width: { xs: 32, md: 40 },
                  height: { xs: 32, md: 40 },
                  objectFit: 'contain'
                }}
              />
            }
            sx={{
              textTransform: 'none',
              fontWeight: 600,
              fontSize: { xs: '0.9rem', md: '1rem' },
              color: location.pathname.includes('/category/kleinanzeigen') || location.pathname === '/' ? 'primary.main' : 'text.secondary',
              borderBottom: location.pathname.includes('/category/kleinanzeigen') || location.pathname === '/' ? '2px solid' : 'none',
              borderColor: 'primary.main',
              borderRadius: 0,
              px: { xs: 2, md: 3 },
              py: { xs: 0.75, md: 1 }, // Reduziert von 1/1.5 auf 0.75/1
              '&:hover': {
                bgcolor: 'transparent',
                color: 'primary.main'
              }
            }}
          >
            Kleinanzeigen
          </Button>
          <Button
            variant="text"
            onClick={() => navigate('/category/autos')}
            startIcon={
              <Box
                component="img"
                src="/images/auto_icon.webp"
                alt="Autos"
                sx={{
                  width: { xs: 40, md: 48 },
                  height: { xs: 40, md: 48 },
                  objectFit: 'contain'
                }}
              />
            }
            sx={{
              textTransform: 'none',
              fontWeight: 600,
              fontSize: { xs: '0.9rem', md: '1rem' },
              color: location.pathname.includes('/category/autos') ? 'primary.main' : 'text.secondary',
              borderBottom: location.pathname.includes('/category/autos') ? '2px solid' : 'none',
              borderColor: 'primary.main',
              borderRadius: 0,
              px: { xs: 2, md: 3 },
              py: { xs: 0.75, md: 1 }, // Reduziert von 1/1.5 auf 0.75/1
              '&:hover': {
                bgcolor: 'transparent',
                color: 'primary.main'
              }
            }}
          >
            Autos
          </Button>
        </Box>
 
        {/* Kategorie-Kärtchen */}
        <CategoryCards theme="kleinanzeigen" />
        
        <Section title="Neueste Kleinanzeigen">
          <Box sx={{ 
            display: 'grid', 
            gridTemplateColumns: { 
              xs: 'repeat(2, 1fr)', // Mobile: 2 Spalten
              sm: 'repeat(2, 1fr)', // Tablet: 2 Spalten
              md: 'repeat(3, 1fr)', // Desktop: 3 Spalten
              lg: 'repeat(4, 1fr)', // Large: 4 Spalten
              xl: 'repeat(5, 1fr)'  // Extra Large: 5 Spalten
            },
            gap: { xs: 2, sm: 2, md: 2.5, lg: 3 }, // Konsistente Abstände
            mt: { xs: 1, sm: 1.5, md: 2 }, // Reduziert von 2/3/3 auf 1/1.5/2
            alignItems: 'start', // Verhindert, dass Karten sich aneinander anpassen
            bgcolor: '#ffffff'
          }}>
            {loadingAds ? (
              Array.from({ length: 6 }).map((_, index) => (
                <AdCardSkeleton key={index} />
              ))
            ) : (
              filteredAds.map((ad) => (
                <AdCard
                  key={ad.id}
                  {...ad}
                />
              ))
            )}
          </Box>
        </Section>
      </Box>
